void mergeSort(int a[], int b[], int low, int high) {
    if (low < high) {
        int mid = (low + high) / 2;
        mergeSort(a, b, low, mid);
        mergeSort(a, b, mid + 1, high);
        combine(a, b, low, mid, high);
    }
}

void combine(int a[], int b[], int low, int mid, int high) {
    int k = low, i = low, j = mid + 1;
    while (i <= mid && j <= high) {
        if (a[i] < a[j]) {
            b[k] = a[k];
            k++;
            i++;
        } else {
            b[k] = a[j];
            j++;
        }
    }
    while (i <= mid) {
        b[k] = a[k];
        k++;
        i++;
    }
    while (j <= high) {
        b[k] = a[k];
        k++;
        j++;
    }
}
