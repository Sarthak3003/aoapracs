   dijkstra(a, v, d);

void dijkstra(int a[6][6], int v[], int d[]) {
   int source = 0;
   int selected = 0;
   int c = 0;
  
   while (c < 6) {
       for (int j = 0; j < 6; j++) {  
           if (v[j] != 1) {   
               if (a[selected][j] > 0) {
                   if ((d[selected] + a[selected][j]) < d[j]) {
                       d[j] = d[selected] + a[selected][j];
                   }
               }  
           }
       }  
 
       for (int i = 0; i < 6; i++) {
           printf("%d\t", d[i]);
       }
       printf("\n");
  
       int min = 100;
       for (int i = 0; i < 6; i++) {
           if (v[i] != 1) {
               if (d[i] < min) {
                   min = d[i];
                   selected = i;  
               }
           }
       }
       v[selected] = 1;
       c++;
   }  
}
