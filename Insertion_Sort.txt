#include <stdio.h> 
#include <time.h>
#include <stdlib.h>

void swap(int *x, int *y) {
	int temp = *y;
	*y = *x;
	*x = temp;
}

void insertionSort(int a[], int n) {
    int i, x, j;
    for (i = 1; i < n; i++) {
        x = a[i];
        j = i - 1;
        while (j >= 0 && a[j] > x) {
            a[j + 1] = a[j];
            j = j - 1;
        }
        a[j + 1] = x;
    }
}

int main() {
	clock_t start_t, end_t, total_t;
	int n = 50000;
	int a[n];
	srand(time(0));
	for(int i = 0; i < n; i++) {
		a[i] = i;
